<script>
  // Navigate to love page when "Yes" is clicked
  document.getElementById('yes').addEventListener('click', () => {
    window.location.href = 'love.html';
  });

  // Make the "No" button run away on hover/touch/click attempts
  const noBtn = document.getElementById('no');

  function moveNo(){
    const padding = 10;
    const w = noBtn.offsetWidth || 120;
    const h = noBtn.offsetHeight || 48;

    // Get viewport size
    const ww = window.innerWidth;
    const wh = window.innerHeight;

    // Compute random position within viewport bounds
    const maxX = Math.max(padding, ww - w - padding);
    const maxY = Math.max(padding, wh - h - padding);
    const x = Math.floor(Math.random() * maxX);
    const y = Math.floor(Math.random() * maxY);

    noBtn.style.left = x + 'px';
    noBtn.style.top  = y + 'px';
    noBtn.style.transform = 'translate(0,0)';
  }

  // Trigger move on different interactions (desktop + mobile)
  ['mouseover','pointerdown','touchstart','focus','click'].forEach(evt=>{
    noBtn.addEventListener(evt, (e)=>{
      e.preventDefault(); // stop any accidental click
      moveNo();
    }, {passive:false});
  });

  // Also reposition if window resizes (keeps it on-screen)
  window.addEventListener('resize', moveNo);

  // Place it somewhere fun on load
  window.addEventListener('load', moveNo);
</script>
